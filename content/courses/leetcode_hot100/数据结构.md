## 1. vector（动态数组）⭐ 宇宙第一

### 📌 基础操作（必须背熟）

```cpp
#include <vector>
using namespace std;

// ===== 创建 =====
vector<int> v1;                    // 空
vector<int> v2(10);                // 10个元素，默认0
vector<int> v3(10, 5);             // 10个元素，都是5
vector<int> v4 = {1, 2, 3, 4, 5};  // 初始化列表
vector<int> v5(v4.begin(), v4.end()); // 拷贝

// ===== 二维数组 =====
vector<vector<int>> matrix(3, vector<int>(4, 0)); // 3行4列全0
vector<vector<int>> mat = {{1,2},{3,4},{5,6}};

// ===== 添加 =====
v1.push_back(10);          // 尾部添加 O(1)
v1.emplace_back(20);       // 更高效（C++11）
v1.insert(v1.begin(), 5);  // 头部插入 O(n)
v1.insert(v1.end(), 3, 99); // 插入3个99

// ===== 删除 =====
v1.pop_back();             // 删除最后一个 O(1)
v1.erase(v1.begin());      // 删除第一个 O(n)
v1.erase(v1.begin() + 1, v1.begin() + 3); // 删除区间
v1.clear();                // 清空

// ===== 访问 =====
int x = v4[0];             // 不检查越界
int y = v4.at(1);          // 检查越界，抛异常
int first = v4.front();    // 第一个
int last = v4.back();      // 最后一个
int* data = v4.data();     // 底层指针

// ===== 容量 =====
int size = v4.size();      // 元素个数
int cap = v4.capacity();   // 容量（≥size）
bool empty = v4.empty();   // 是否为空
v4.reserve(100);           // 预留空间（不改变size）
v4.resize(10);             // 调整size（多出的默认0）
v4.shrink_to_fit();        // 释放多余内存

// ===== 排序 =====
sort(v4.begin(), v4.end());              // 升序
sort(v4.begin(), v4.end(), greater<int>()); // 降序
sort(v4.begin(), v4.end(), [](int a, int b) {
    return a > b;  // 自定义降序
});

// ===== 查找 =====
auto it = find(v4.begin(), v4.end(), 3); // 找到返回迭代器，否则end()
int idx = it - v4.begin();  // 索引

// ===== 遍历 =====
for (int i = 0; i < v4.size(); i++) { }
for (auto it = v4.begin(); it != v4.end(); it++) { }
for (int x : v4) { }
for (auto& x : v4) { x *= 2; } // 修改元素
for (const auto& x : v4) { }   // 只读，避免拷贝
```

## 2. unordered_map（哈希字典）⭐ 查找神器

### 📌 基础操作

```cpp
#include <unordered_map>

// ===== 创建 =====
unordered_map<string, int> m1;
unordered_map<int, int> m2 = {{1, 10}, {2, 20}, {3, 30}};

// ===== 插入 =====
m1["apple"] = 5;                    // 直接赋值
m1.insert({"banana", 3});           // insert
m1.insert(make_pair("orange", 7));
m1.emplace("grape", 2);             // emplace更高效

// ===== 访问 =====
int val = m1["apple"];              // 不存在会插入默认值0！
int val2 = m1.at("banana");         // 不存在抛异常（推荐）
// 安全访问：
if (m1.count("apple")) {
    cout << m1["apple"] << endl;
}

// ===== 查找 =====
if (m1.find("apple") != m1.end()) {
    cout << "找到了：" << m1["apple"] << endl;
}

if (m1.count("apple") > 0) {        // count返回0或1
    cout << "存在" << endl;
}

// ===== 删除 =====
m1.erase("apple");                  // 删除key
m1.erase(m1.begin());               // 删除第一个

// ===== 遍历 =====
for (auto& p : m1) {
    cout << p.first << " → " << p.second << endl;
}

// C++17 结构化绑定
for (auto& [key, value] : m1) {
    cout << key << " → " << value << endl;
}

// ===== 大小 =====
int size = m1.size();
bool empty = m1.empty();
m1.clear();
```

### 🔥 算法题经典场景

```cpp
// 5. 前缀和 + 哈希（子数组问题）

prefix_sum[0] = 1;

for (int num : nums) {
    sum += num;
```

```cpp
// 4. 逆波兰表达式

    } else if (token == "-") {
        // ...
    } else {
        st.push(stoi(token));
    }
}

// 5. 字符串解码（3[a]2[bc] → aaabcbc）
stack<string> str;
stack<int> num;

        num.push(c - '0');

        if (!str.empty()) str.top() += c;
        else str.push(string(1, c));
```

## 6. priority_queue（优先队列）⭐ TopK、贪心

### 📌 基础操作

```cpp
#include <queue>  // priority_queue 也在 queue 头文件

// ===== 默认大顶堆 =====
priority_queue<int> pq;
pq.push(3);
pq.push(1);
pq.push(5);
int top = pq.top(); // → 5（最大值）
pq.pop();           // 删除5

// ===== 小顶堆 =====
priority_queue<int, vector<int>, greater<int>> min_heap;
min_heap.push(3);
min_heap.push(1);
min_heap.push(5);
int top2 = min_heap.top(); // → 1（最小值）

// ===== 自定义比较（大顶堆） =====
auto cmp = [](int a, int b) { return a < b; };
priority_queue<int, vector<int>, decltype(cmp)> pq2(cmp);

// ===== 自定义结构体 =====
struct Node {
    int val, idx;
    bool operator<(const Node& other) const {
        return val < other.val;  // 大顶堆
    }
};
priority_queue<Node> pq3;

// 或者（C++17）
struct Compare {
    bool operator()(const Node& a, const Node& b) {
        return a.val < b.val;  // 大顶堆
    }
};
priority_queue<Node, vector<Node>, Compare> pq4;
```

```cpp
// 1. 回文串
bool isPalindrome(string s) {
    int l = 0, r = s.size() - 1;
    while (l < r) {
        if (s[l] != s[r]) return false;
        l++; r--;
    }
    return true;
}

// 2. 字符串反转
reverse(s.begin(), s.end());

// 3. 字符频率
vector<int> count(26, 0);
for (char c : s) count[c - 'a']++;

// 4. 字符串压缩
string compress(string s) {
    string result;
    int i = 0;
    while (i < s.size()) {
        int j = i;
        while (j < s.size() && s[j] == s[i]) j++;
        result += s[i] + to_string(j - i);
        i = j;
    }
    return result.size() < s.size() ? result : s;
}
```

## 10. 算法题常用技巧合集

### 📦 容器转换

```cpp
// vector → set
unordered_set<int> s(nums.begin(), nums.end());

// set → vector
vector<int> v(s.begin(), s.end());

// vector → 字符串（数字）

// 字符串 → vector

```

### 🔄 迭代器技巧

```cpp
// 反向遍历
for (auto it = v.rbegin(); it != v.rend(); it++) { }

// 获取索引
int idx = find(v.begin(), v.end(), target) - v.begin();

// 删除符合条件的元素（erase-remove 惯用法）
```

## 🎯 面试高频题型对应容器

| 题型     | 推荐容器              | 原因     |
| -------- | --------------------- | -------- |
| 数组操作 | `vector`              | 万能     |
| 哈希表   | `unordered_map`       | O(1)查找 |
| 去重     | `unordered_set`       | O(1)     |
| BFS      | `queue`               | 先进先出 |

| 字符计数 | `vector<int>(26)`     | 更高效   |

| 图       | `vector<vector<int>>` | 邻接表   |
