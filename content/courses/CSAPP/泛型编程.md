## 第一部分：内存安全与系统安全伦理

### 1.1 Use-After-Free 漏洞深度解析

**Use-After-Free** 是指在堆内存已被释放后，仍然继续使用该内存的错误。这是C语言中最常见也最危险的内存错误之一。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    // 分配堆内存
    char *bytes = malloc(4);
    char *ptr = bytes;

    // 正常使用内存
    strcpy(bytes, "Hi!");
    printf("Before free: %s\n", bytes);

    // 释放内存
    free(bytes);

    // ❌ 危险：使用已释放的内存
    // bytes 已被释放，但 ptr 仍然指向该地址
    // 这块内存可能已被重新分配用于其他用途
    strncpy(ptr, "Oh!", 3);  // Use-After-Free 漏洞！

    // 可能输出乱码、崩溃，或"正常工作"（最危险的情况）
    printf("After free: %s\n", ptr);

    // 更危险的场景：内存被复用
    int *num = malloc(sizeof(int));
    *num = 42;
    // 此时 ptr 指向的内存可能已被 num 占用
    printf("Value at ptr: %d\n", *(int *)ptr);  // 可能打印 42！

    free(num);
    // 注意：不能 free(ptr) 或 free(bytes)，会导致 double free

    return 0;
}
```

**内存复用示意图：**

```
时间线：
────────────────────────────────────────────────────────────────►

步骤1: malloc(4)                   步骤2: free(bytes)
┌─────────────┐                    ┌─────────────┐
│   bytes     │                    │   bytes     │
│     ↓       │                    │     ↓       │
│ ┌─────────┐ │      free()       │ ┌─────────┐ │
│ │ "Hi!\0"  │ │ ─────────────►   │ │ ???????  │ │  (内存被标记为空闲)
│ └─────────┘ │                    │ └─────────┘ │
└─────────────┘                    └─────────────┘

步骤3: malloc(sizeof(int))         步骤4: 通过 ptr 访问
┌─────────────┐                    ┌─────────────┐
│   num       │                    │   ptr (悬空) │
│     ↓       │                    │     ↓       │
│ ┌─────────┐ │                    │ ┌─────────┐ │
│ │   42    │ │  (复用同一块内存)   │ │   42    │ │
│ └─────────┘ │                    │ └─────────┘ │
└─────────────┘                    └─────────────┘
                                   ptr 现在访问的是 num 的数据！
```

> **关键洞察**：`free()` 不会修改传入的指针值，它只是释放指针所指向的内存。指针本身仍然保存着原来的地址，这就是为什么我们可以继续（错误地）使用它。

## 第二部分：泛型编程基础（void* 与内存操作）

### 2.1 void* 核心概念

`void*` 是一个**通用指针类型**，可以指向任何类型的数据。

```c
#include <stdio.h>
#include <string.h>

int main() {
    int x = 42;
    double d = 3.14;
    char c = 'A';

    void *generic_ptr;

    // void* 可以指向任何类型
    generic_ptr = &x;
    printf("Pointing to int\n");

    generic_ptr = &d;
    printf("Pointing to double\n");

    generic_ptr = &c;
    printf("Pointing to char\n");

    // ❌ 不能解引用 void*（编译器不知道类型大小）
    // printf("%d\n", *generic_ptr);  // 编译错误

    // ✅ 需要先转换回具体类型
    printf("Value: %c\n", *(char *)generic_ptr);

    return 0;
}
```

**void* 的关键限制：**

| 操作           | 是否允许 | 原因           |
| -------------- | -------- | -------------- |
| 赋值给其他指针 | ✅ 允许   | 可隐式转换     |
| 转换为具体类型 | ✅ 允许   | 需要显式转换   |
| 解引用         | ❌ 不允许 | 不知道类型大小 |
| 指针算术       | ❌ 不允许 | 不知道步长     |
| 比较           | ✅ 允许   | 比较地址值     |

### 2.2 memset、memcpy 和 memmove

```c
#include <stdio.h>
#include <string.h>

/**
 * memset - 设置内存区域为指定字节值
 * void *memset(void *s, int c, size_t n);
 * 将 s 指向的内存区域的 n 个字节设为 c
 */

/**
 * memcpy - 复制内存块（不支持重叠）
 * void *memcpy(void *dest, const void *src, size_t n);
 */

/**
 * memmove - 复制内存块（支持重叠）
 * void *memmove(void *dest, const void *src, size_t n);
 */

int main() {
    // memset 示例
    int counts[5];
    memset(counts, 0, sizeof(counts));  // 全部清零

    // 危险：memset 是按字节设置的！
    int arr[5];
    memset(arr, 1, sizeof(arr));
    // 每个字节被设为 0x01，所以每个 int 变成 0x01010101，不是 1！

    // memcpy 示例
    int x = 5, y = 4;
    memcpy(&x, &y, sizeof(x));
    printf("After memcpy: x=%d, y=%d\n", x, y);

    // memmove 示例（重叠区域）
    int nums[] = {1, 2, 3, 4, 5};
    // 将 nums[0..2] 移动到 nums[1..3]（重叠区域）
    memmove(nums + 1, nums, 3 * sizeof(int));
    // nums 现在: {1, 1, 2, 3, 5}

    return 0;
}
```

**memcpy vs memmove：**

| 特性     | memcpy                     | memmove      |
| -------- | -------------------------- | ------------ |
| 重叠区域 | ❌ 不支持（未定义行为）     | ✅ 支持       |
| 性能     | 通常更快（无重叠检查开销） | 可能稍慢     |
| 使用场景 | 确信区域不重叠             | 区域可能重叠 |
| 默认选择 | 优先使用（表达意图清晰）   | 需要时使用   |

### 2.3 泛型交换函数

```c
#include <stdio.h>
#include <string.h>

/**
 * 泛型交换函数 - 可以交换任意类型的两个值
 * @param a: 指向第一个数据的指针
 * @param b: 指向第二个数据的指针
 * @param nbytes: 每个数据类型的字节数
 */
void swap(void *a, void *b, size_t nbytes) {
    // 在栈上分配临时存储空间（VLA - 变长数组）
    char temp[nbytes];

    // 步骤1: 保存 a 到 temp
    memcpy(temp, a, nbytes);

    // 步骤2: 复制 b 到 a 的位置
    memcpy(a, b, nbytes);

    // 步骤3: 复制 temp 到 b 的位置
    memcpy(b, temp, nbytes);
}

int main() {
    // 交换 int
    int x = 2, y = 5;
    printf("Before int swap: x=%d, y=%d\n", x, y);
    swap(&x, &y, sizeof(x));
    printf("After int swap:  x=%d, y=%d\n\n", x, y);

    // 交换 double
    double d1 = 3.14, d2 = 2.718;
    printf("Before double swap: d1=%f, d2=%f\n", d1, d2);
    swap(&d1, &d2, sizeof(d1));
    printf("After double swap:  d1=%f, d2=%f\n\n", d1, d2);

    // 交换字符串指针（交换的是指针，不是字符串内容）
    char *str1 = "Hello", *str2 = "World";
    printf("Before string swap: str1=%s, str2=%s\n", str1, str2);
    swap(&str1, &str2, sizeof(str1));
    printf("After string swap:  str1=%s, str2=%s\n", str1, str2);

    return 0;
}
```

### 2.4 void* 的危险性

```c
#include <stdio.h>
#include <string.h>

void dangerous_swap(void *a, void *b, size_t n) {
    char temp[n];
    memcpy(temp, a, n);
    memcpy(a, b, n);
    memcpy(b, temp, n);
}

int main() {
    // 危险示例1：只交换 int 的一半
    int x = 0xFFFFFFFF;  // 所有位都是1
    int y = 0xEEEEEEEE;  // 交替的1和0
    printf("Before: x=0x%x, y=0x%x\n", x, y);

    dangerous_swap(&x, &y, sizeof(short));  // 只交换2字节！
    printf("After:  x=0x%x, y=0x%x (corrupted!)\n", x, y);

    // 危险示例2：类型大小不匹配
    int a = 100;
    short b = 200;
    // dangerous_swap(&a, &b, sizeof(short));  // 未定义行为！

    // 危险示例3：混合类型
    int c = 42;
    double d = 3.14;
    // dangerous_swap(&c, &d, sizeof(int));  // 只复制部分 double

    return 0;
}
```

> **核心教训**：void* 放弃了类型检查，程序员必须确保大小和类型匹配！

## 第三部分：函数指针基础

### 3.1 为什么需要函数指针？

函数指针允许我们将**行为/逻辑**作为参数传递给函数，从而实现高度可定制的泛型代码。

```c

// 解决方案：让调用者传入比较函数！

// 比较函数类型：接受两个 int，返回是否需要交换
typedef bool (*compare_fn)(int, int);

void bubble_sort_int(int *arr, size_t n, compare_fn should_swap) {
    // 实现使用 should_swap 来决定是否交换
}
```

### 3.2 函数指针声明语法

```c
#include <stdio.h>
#include <stdbool.h>

// 函数指针的声明语法：
// [返回类型] (*[指针名])([参数类型列表])

// 示例1：指向返回 bool、接受两个 int 的函数
bool (*should_swap)(int, int);

// 示例2：指向返回 int、接受一个 double 的函数
int (*func_ptr)(double);

// 示例3：指向返回 void、接受一个 char* 的函数
void (*log_func)(char *);

// 示例4：使用 typedef 简化
typedef bool (*CompareFunc)(int, int);

// 现在可以像使用普通类型一样使用 CompareFunc
void sort(int *arr, size_t n, CompareFunc cmp) {
    // 实现...
}

// 实际比较函数
bool ascending(int a, int b) { return a > b; }
bool descending(int a, int b) { return a < b; }

int main() {
    int nums[] = {4, 2, 5, 1, 3};

    // 函数名就是函数指针
    CompareFunc cmp = ascending;

    // 通过函数指针调用
    bool result = cmp(5, 3);  // 等价于 ascending(5, 3)
    printf("5 > 3? %s\n", result ? "true" : "false");

    return 0;
}
```

### 3.3 函数指针类型解析

| 声明                            | 含义                                                        |
| ------------------------------- | ----------------------------------------------------------- |
| `bool (*should_swap)(int, int)` | should_swap 是指向【返回 bool，接受两个 int】的函数的指针   |
| `int (*cmp)(void*, void*)`      | cmp 是指向【返回 int，接受两个 void*】的函数的指针          |
| `void (*callback)(char*, int)`  | callback 是指向【返回 void，接受 char* 和 int】的函数的指针 |
| `int (*ops[])(int, int)`        | ops 是【指向返回 int、接受两个 int 的函数】的指针数组       |

## 第四部分：完全泛型冒泡排序

### 4.1 标准比较函数签名

C 标准库使用的比较函数格式：

```c

 * @param a: 指向第一个元素的指针

 * @return:
 *   - 负数：a 应该排在 b 之前

 *   - 正数：a 应该排在 b 之后
 */

```

### 4.2 泛型冒泡排序完整实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

// 泛型交换函数
void swap(void *a, void *b, size_t nbytes) {
    char temp[nbytes];
    memcpy(temp, a, nbytes);
    memcpy(a, b, nbytes);
    memcpy(b, temp, nbytes);
}

/**
 * 泛型数组元素访问：
 * 获取第 i 个元素的地址 = (char *)base + i * elem_size
 */

/**
 * 完全泛型冒泡排序
 * @param base: 数组起始地址
 * @param nelems: 元素个数
 * @param elem_size: 每个元素的字节数
 * @param cmp: 比较函数（返回负数/0/正数）
 */
void bubble_sort(void *base, size_t nelems, size_t elem_size,
                 int (*cmp)(void *, void *)) {
    while (true) {
        bool swapped = false;
        for (size_t i = 1; i < nelems; i++) {
            // 获取第 i-1 和 i 个元素的指针（字节算术）
            void *prev = (char *)base + (i - 1) * elem_size;
            void *curr = (char *)base + i * elem_size;

            // 如果需要交换（前一个 > 当前）
            if (cmp(prev, curr) > 0) {
                swapped = true;
                swap(prev, curr, elem_size);
            }
        }
        if (!swapped) {
            return;
        }
    }
}

// ============ 各种类型的比较函数 ============

// 整数比较
int cmp_int_ascending(void *a, void *b) {
    int ia = *(int *)a;
    int ib = *(int *)b;
    return ia - ib;  // 负数表示 ia < ib
}

int cmp_int_descending(void *a, void *b) {
    int ia = *(int *)a;
    int ib = *(int *)b;
    return ib - ia;
}

// 绝对值比较
int cmp_int_abs(void *a, void *b) {
    int ia = *(int *)a;
    int ib = *(int *)b;
    int abs_a = ia < 0 ? -ia : ia;
    int abs_b = ib < 0 ? -ib : ib;
    return abs_a - abs_b;
}

// 字符串比较（注意：需要双重解引用）
int cmp_string_lexicographic(void *a, void *b) {
    // a 和 b 指向 char* 元素，所以类型是 char**
    char *str_a = *(char **)a;
    char *str_b = *(char **)b;
    return strcmp(str_a, str_b);
}

int cmp_string_length(void *a, void *b) {
    char *str_a = *(char **)a;
    char *str_b = *(char **)b;
    size_t len_a = strlen(str_a);
    size_t len_b = strlen(str_b);
    return (len_a > len_b) - (len_a < len_b);
}

// double 比较
int cmp_double(void *a, void *b) {
    double da = *(double *)a;
    double db = *(double *)b;
    if (da < db) return -1;
    if (da > db) return 1;
    return 0;
}

// ============ 辅助函数 ============

void print_int_array(int arr[], size_t n) {
    printf("[");
    for (size_t i = 0; i < n; i++) {
        printf("%d", arr[i]);
        if (i < n - 1) printf(", ");
    }
    printf("]\n");
}

void print_string_array(char *arr[], size_t n) {
    printf("[");
    for (size_t i = 0; i < n; i++) {
        printf("\"%s\"", arr[i]);
        if (i < n - 1) printf(", ");
    }
    printf("]\n");
}

// ============ 测试代码 ============

int main() {
    // 测试整数数组
    printf("=== Integer Array ===\n");
    int nums[] = {5, -3, 8, 1, -7, 4};
    size_t n_int = sizeof(nums) / sizeof(nums[0]);

    int nums_copy[6];
    memcpy(nums_copy, nums, sizeof(nums));
    printf("Original:  ");
    print_int_array(nums_copy, n_int);

    bubble_sort(nums_copy, n_int, sizeof(int), cmp_int_ascending);
    printf("Ascending: ");
    print_int_array(nums_copy, n_int);

    memcpy(nums_copy, nums, sizeof(nums));
    bubble_sort(nums_copy, n_int, sizeof(int), cmp_int_descending);
    printf("Descending:");
    print_int_array(nums_copy, n_int);

    memcpy(nums_copy, nums, sizeof(nums));
    bubble_sort(nums_copy, n_int, sizeof(int), cmp_int_abs);
    printf("Abs sort:  ");
    print_int_array(nums_copy, n_int);

    // 测试字符串数组
    printf("\n=== String Array ===\n");
    char *fruits[] = {"banana", "apple", "cherry", "date", "elderberry"};
    size_t n_str = sizeof(fruits) / sizeof(fruits[0]);

    char *fruits_copy[5];
    memcpy(fruits_copy, fruits, sizeof(fruits));
    printf("Original:     ");
    print_string_array(fruits_copy, n_str);
```

```c
 * 统计数组中满足条件的元素个数
 */
int count_matches(void *base, size_t nelems, size_t elem_size,
                  bool (*match_fn)(void *)) {
    int count = 0;
    for (size_t i = 0; i < nelems; i++) {
        void *elem = (char *)base + i * elem_size;
        if (match_fn(elem)) {
            count++;
        }
    }
    return count;
```

## 第六部分：字符串比较的深层理解

### 6.1 指针层级分析

对于字符串数组，理解指针的层级至关重要：

```c
// 字符串数组的内存布局
char *words[] = {"apple", "banana", "cherry"};

// words 是一个数组，每个元素是 char*
//
// 内存示意图（64位系统）：
//
// 地址（低）                         地址（高）
// ┌────────┬────────┬────────┐
// │ word0  │ word1  │ word2  │  ← words 数组（每个元素 8 字节）
// │ 0x1000 │ 0x2000 │ 0x3000 │
// └───┬────┴───┬────┴───┬────┘
//     │        │        │
//     v        v        v
// ┌────────┐ ┌────────┐ ┌────────┐
// │ 'a'    │ │ 'b'    │ │ 'c'    │  ← 字符串数据
// │ 'p'    │ │ 'a'    │ │ 'h'    │    （可能在只读数据段）
// │ 'p'    │ │ 'n'    │ │ 'e'    │
// │ 'l'    │ │ 'a'    │ │ 'r'    │
// │ 'e'    │ │ 'n'    │ │ 'r'    │
// │ '\0'   │ │ 'a'    │ │ 'y'    │
// │        │ │ '\0'   │ │ '\0'   │
// └────────┘ └────────┘ └────────┘

// 当 bubble_sort 调用比较函数时：
// cmp(prev, curr) 中 prev 和 curr 指向数组元素
// 所以 prev 的类型是 char**（指向 char* 的指针）
```

### 6.2 正确的字符串比较函数实现

```c
#include <stdio.h>
#include <string.h>

/**
 * 字符串比较函数 - 正确实现
 *
 * 关键理解：
 * - 传入的 void* 指向数组元素
 * - 数组元素类型是 char*，所以实际参数类型是 char**
 * - 需要双重解引用才能得到实际字符串
 */

// 正确做法：逐步解引用
int cmp_string_correct(void *a, void *b) {
    // 步骤1：转换为正确的指针类型 char**
    char **ptr_to_str_a = (char **)a;
    char **ptr_to_str_b = (char **)b;

    // 步骤2：解引用得到实际字符串 char*
    char *str_a = *ptr_to_str_a;
    char *str_b = *ptr_to_str_b;

    // 步骤3：使用 strcmp 比较
    return strcmp(str_a, str_b);
}

// 简化版本（一行）
int cmp_string_compact(void *a, void *b) {
    return strcmp(*(char **)a, *(char **)b);
}

// 错误版本1：忘记解引用
int cmp_string_wrong1(void *a, void *b) {
    // 错误：把地址当作字符串，a 是 char**，不是 char*
    return strcmp((char *)a, (char *)b);
}

// 错误版本2：多了解引用一次
int cmp_string_wrong2(void *a, void *b) {
    // 错误：解引用太多次，实际想访问字符串内容
    // 但 a 是指向 char* 的指针，三次解引用会出错
    char *str_a = **(char ***)a;  // 错误！
    char *str_b = **(char ***)b;
    return strcmp(str_a, str_b);
}

// 按字符串长度比较
int cmp_string_length(void *a, void *b) {
    char *str_a = *(char **)a;
    char *str_b = *(char **)b;
    size_t len_a = strlen(str_a);
    size_t len_b = strlen(str_b);
    return (len_a > len_b) - (len_a < len_b);
}

int main() {
    char *fruits[] = {"banana", "apple", "cherry"};
    size_t n = sizeof(fruits) / sizeof(fruits[0]);

    printf("Before: %s, %s, %s\n", fruits[0], fruits[1], fruits[2]);

    bubble_sort(fruits, n, sizeof(char *), cmp_string_correct);

    printf("After:  %s, %s, %s\n", fruits[0], fruits[1], fruits[2]);

    return 0;
}
```
